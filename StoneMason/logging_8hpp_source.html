116  static void consumeLogs() {
117 
118  getLogQueueMtx().lock();
119 
120  bool empty = getLogQueue().empty();
121  if (!empty) {
122  LogRecord top = std::move(getLogQueue().front());
123  getLogQueue().pop();
124 
125  getLogQueueMtx().unlock();
126 
127  // formatting error: is `top.msg.data()` properly null-terminated?
128  fmt::memory_buffer logMsg;
129 
130  fmt::memory_buffer fileUrl;
131  format_to(fileUrl, "file://{}:{}", top.file, top.line);
132 
133  time_t localtimeReady = std::chrono::system_clock::to_time_t(top.time);
134  auto seconds = std::chrono::time_point_cast<std::chrono::seconds>(top.time);
135  auto ms = std::chrono::duration_cast<std::chrono::nanoseconds>(top.time - seconds);
136 
137  format_to(logMsg, "[{0:%T}.{1:<12}] [{2:^72}] [{3:<8}]: {4}", *std::localtime(&localtimeReady),
138  ms.count(),
139  to_string(fileUrl), logLevelToString(top.level), to_string(top.msg));
140 
141  std::string finalMsg = to_string(logMsg); // don't flush!
142 
143  for (const auto &func : getLogHooks()) {
144  func(&top, &finalMsg);
145  }
146 
147  // Recurse. No need to check/set the consume flag as they are only modified on exit/enter.
148  // This is better than just looping bc it breaks the consume task up into multiple submits
149  // to the thread pool!
150  if (getLogThreadPool() != nullptr) {
151  getLogThreadPool()->submitTask(consumeLogs);
152  } else {
153  consumeLogs();
154  }
155  } else {
156  getConsumingFlag() = false;
157 
158  getLogQueueMtx().unlock();
159  }
160  }
